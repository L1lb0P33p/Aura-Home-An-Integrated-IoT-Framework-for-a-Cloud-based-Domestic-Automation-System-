#include "thingProperties.h"



void setup()
{
  Serial.begin(9600);
  Serial.begin(9600,SERIAL_8N1);
  pinMode(2,OUTPUT);
  delay(1500);

  initProperties();

  ArduinoCloud.begin(ArduinoIoTPreferredConnection);

  setDebugMessageLevel(2);
  ArduinoCloud.printDebugInfo();
}

void loop()
{
  ArduinoCloud.update();
}

/************************************************
            BULB 1
************************************************/
void onBulb1Change()
{
  if(BULB1)
  {
    
    Serial.write('A');      // Bulb1 ON
  }
  else
  {
    
    Serial.write('a');      // Bulb1 OFF
  }
}
